<script>

function show(id){

document.querySelectorAll(".section").forEach(s=>{
s.classList.remove("active")
})

document.getElementById(id).classList.add("active")

}


function convert(){

let v=document.getElementById("valorant").value

let cs=(v*3.18).toFixed(2)

document.getElementById("convertResult").innerText=
"CS2 Sens: "+cs

}


function calcEDPI(){

let dpi=document.getElementById("dpi").value
let sens=document.getElementById("sens").value

let edpi=dpi*sens

document.getElementById("edpiResult").innerText=
"eDPI: "+edpi

}


function calcCM(){

let dpi=document.getElementById("dpi2").value
let sens=document.getElementById("sens2").value

let cm=(2.54*360)/(dpi*sens)

document.getElementById("cmResult").innerText=
"cm/360: "+cm.toFixed(2)

}


function addPlayer(){

let name=document.getElementById("name").value
let dpi=document.getElementById("dpi3").value
let sens=document.getElementById("sens3").value

let card=document.createElement("div")

card.className="card"

card.innerHTML=
"<b>"+name+"</b><br>DPI: "+dpi+"<br>Sens: "+sens

document.getElementById("playersList").appendChild(card)

}

</script>
